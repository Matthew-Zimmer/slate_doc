Include File: meta/meta.hpp

Fully Qualified Name: ::Slate::Meta::Args

Declaration


    template <typename Type>
    using Args;
is_template: true

Abstract

Results in a meta container of the argument types of a function pointer

Example


    include <iostream>
    include <meta/meta.hpp>
    
    using Slate::Meta::Wrap;
    using Slate::Meta::Args;
    
    int f(int x, double h)
    {}
    
    class A
    {
    public: 
      double g() const
      {}
    };
    
    int main()
    {
      //formats bools as true / false
      std::cout << std::boolalpha << std::is_same_v<Args<decltype(&f)>, Wrap<int, double>> << std::endl;
      std::cout << std::boolalpha << std::is_same_v<Args<decltype(&A::g)>, Wrap<>> << std::endl;
    }

Output


    true
    true
