Include File: meta/meta.hpp

Fully Qualified Name: ::Slate::Meta::Return_Type

Declaration


    template <typename Type>
    using Return_Type;
is_template: true

Abstract

Results in the return type of a function pointer

Example


    include <iostream>
    include <meta/meta.hpp>
    
    using Slate::Meta::Return_Type;
    
    int f()
    {}
    
    class A
    {
    public: 
      double g() const
      {}
    };
    
    int main()
    {
      //formats bools as true / false
      std::cout << std::boolalpha << std::is_same_v<Return_Type<decltype(&f)>, int> << std::endl;
      std::cout << std::boolalpha << std::is_same_v<Return_Type<decltype(&A::g)>, double> << std::endl;
    }

Output


    true
    true
