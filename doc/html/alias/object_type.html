Include File: meta/meta.hpp

Fully Qualified Name: ::Slate::Meta::Object_Type

Declaration


    template <typename Type>
    using Object_Type;
is_template: true

Abstract

Results in the object type of a function pointer

Example


    include <iostream>
    include <meta/meta.hpp>
    
    using Slate::Meta::Object_Type;
    
    class A
    {
    public: 
      double g() const
      {}
    };
    
    int main()
    {
      //formats bools as true / false
      std::cout << std::boolalpha << std::is_same_v<Object_Type<decltype(&A::g)>, A const> << std::endl;
    }

Output


    true
